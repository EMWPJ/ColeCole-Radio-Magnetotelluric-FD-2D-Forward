## 2. 理论基础

### 2.1 Cole-Cole频散模型

单Cole-Cole极化模型的复电导率：

$$\sigma^*(\omega) = \sigma_0 \left[1 + m \left(1 - \frac{1}{1 + (i\omega\tau)^c}\right)\right]$$

其中：
- $\sigma_0$: 静态电导率 (S/m)
- $m$: 极化强度 (0 ≤ m ≤ 1)
- $\tau$: 时间常数 (秒)
- $c$: Cole-Cole指数 (0 ≤ c ≤ 1)

### 2.2 含位移电流的总电导率

$$\sigma_{total}^* = \sigma^* + i\omega\varepsilon_0\varepsilon_r$$

### 2.3 控制方程

**TE模式**（电场Ex分量）：

$$-\frac{\partial}{\partial x}\left(\frac{1}{i\omega\mu}\frac{\partial E_x}{\partial x}\right) - \frac{\partial}{\partial z}\left(\frac{1}{i\omega\mu}\frac{\partial E_x}{\partial z}\right) + \sigma^* E_x = 0$$

**TM模式**（磁场Hx分量）：

$$-\frac{\partial}{\partial x}\left(\frac{1}{\sigma^*}\frac{\partial H_x}{\partial x}\right) - \frac{\partial}{\partial z}\left(\frac{1}{\sigma^*}\frac{\partial H_x}{\partial z}\right) + i\omega\mu H_x = 0$$

### 2.4 阻抗与视电阻率

$$Z_{xy} = \frac{E_x}{H_y}, \quad Z_{yx} = \frac{E_y}{H_x}$$

$$\rho_a = \frac{|Z|^2}{\omega\mu}, \quad \phi = \arg(Z)$$

### 4.2 Cole-Cole模型 (`core/cole_cole.py`)

```python
import numpy as np

class ColeColeSingle:
    """
    单Cole-Cole极化模型
    
    sigma*(omega) = sigma0 * [1 + m * (1 - 1/(1 + (i*omega*tau)^c))]
    """
    
    def __init__(self, sigma0: float, m: float, tau: float, c: float):
        """
        参数:
            sigma0: 静态电导率 (S/m)
            m: 极化强度 (0~1)
            tau: 时间常数 (秒)
            c: Cole-Cole指数 (0~1)
        """
        self.sigma0 = sigma0
        self.m = m
        self.tau = tau
        self.c = c
    
    def sigma(self, omega: float) -> complex:
        """计算Cole-Cole复电导率"""
        v = (1j * omega * self.tau) ** self.c
        denom = 1.0 + v
        factor = 1.0 + self.m * (1.0 - 1.0 / denom)
        return self.sigma0 * factor
    
    def sigma_total(self, omega: float, epsilon_r: float = 1.0) -> complex:
        """含位移电流的总电导率"""
        eps0 = 8.854e-12
        return self.sigma(omega) + 1j * omega * eps0 * epsilon_r
```

### 4.3 网格模块 (`core/mesh.py`)

```python
import numpy as np

class RectGrid256:
    """
    256×256非交错矩形网格
    
    节点编号: (j, k) -> n = k * (nx+1) + j
    电导率数组: sigma[k, j] 对应 sigma(j, k) in Fortran
    """
    
    def __init__(self, nx=256, nz=256, pad=20):
        self.nx = nx
        self.nz = nz
        self.pad = pad
        self.core_cols = nx - 2 * pad  # 216
        self.nnodes = (nz + 1) * (nx + 1)  # 257×257 = 66049
        
        # 变步长数组
        self.dy = self._generate_dy()  # shape (256,)
        self.dz = self._generate_dz()  # shape (256,)
        
        # 累计坐标
        self.y_nodes = np.concatenate([[0], np.cumsum(self.dy)])
        self.z_nodes = np.concatenate([[0], np.cumsum(self.dz)])
    
    def _generate_dy(self):
        """y方向变步长：中间密、两端疏"""
        dy = np.zeros(self.nx)
        # 中心区域
        center_start = self.pad
        center_end = self.nx - self.pad
        center_dy = 50.0  # 中心步长约50m
        
        for j in range(self.nx):
            if j < center_start:
                # 左侧边界加宽
                factor = 1.0 + (center_start - j) * 0.1
                dy[j] = center_dy * factor
            elif j > center_end:
                # 右侧边界加宽
                factor = 1.0 + (j - center_end) * 0.1
                dy[j] = center_dy * factor
            else:
                dy[j] = center_dy
        return dy
    
    def _generate_dz(self):
        """z方向变步长：浅层密、深层疏"""
        dz = np.zeros(self.nz)
        dz0 = 20.0  # 浅层基础步长
        growth = 1.05  # 增长因子
        
        cumsum = 0
        for k in range(self.nz):
            dz[k] = dz0 * (growth ** k)
            cumsum += dz[k]
        
        # 归一化确保总深度
        total_depth = cumsum
        dz = dz / cumsum * total_depth
        return dz
    
    def node_index(self, j: int, k: int) -> int:
        """(j, k) -> 一维节点编号"""
        return k * (self.nx + 1) + j
    
    def sigma_hat(self, sigma):
        """
        变步长加权平均电导率
        
        对应Fortran lines 50-58:
        sigma_hat(J1,K1) = weighted_average(sigma, dy, dz)
        """
        sh = np.zeros((self.nz + 1, self.nx + 1), dtype=complex)
        
        # 边界
        sh[0, :] = sigma[0, :]
        sh[self.nz, :] = sigma[self.nz - 1, :]
        sh[:, 0] = sigma[:, 0]
        sh[:, self.nx] = sigma[:, self.nx - 1]
        
        # 内部加权平均
        for j in range(1, self.nx):
            for k in range(1, self.nz):
                w1 = self.dy[j-1] * self.dz[k-1]
                w2 = self.dy[j] * self.dz[k-1]
                w3 = self.dy[j-1] * self.dz[k]
                w4 = self.dy[j] * self.dz[k]
                wt = w1 + w2 + w3 + w4
                
                sh[k, j] = (sigma[k-1, j-1] * w1 + 
                             sigma[k-1, j] * w2 + 
                             sigma[k, j-1] * w3 + 
                             sigma[k, j] * w4) / wt
        return sh
```

### 4.4 有限差分算子 (`core/fd_operator.py`)

```python
import numpy as np
from scipy.sparse import csr_matrix

def add_sparse_entry(rows, cols, data, r, c, v):
    """添加稀疏矩阵元素"""
    rows.append(r)
    cols.append(c)
    data.append(v)

def assemble_te_matrix(grid, sigma_hat, omega, mu=np.pi*4e-7):
    """
    组装TE模式五点差分刚度矩阵
    
    对应Fortran MT2DTE子程序
    
    方程: (1/iωμ)∇²Eₓ + σ*Eₓ = 0
    """
    n = grid.nnodes
    rows, cols, data = [], [], []
    
    for k in range(1, grid.nz):
        for j in range(1, grid.nx):
            idx = grid.node_index(j, k)
            
            # 五点差分系数
            c_n = 2.0 / (grid.dz[k-1] * (grid.dz[k-1] + grid.dz[k]))
            c_s = 2.0 / (grid.dz[k] * (grid.dz[k-1] + grid.dz[k]))
            c_w = 2.0 / (grid.dy[j-1] * (grid.dy[j-1] + grid.dy[j]))
            c_e = 2.0 / (grid.dy[j] * (grid.dy[j-1] + grid.dy[j]))
            
            # 中心系数（含源项）
            c_center = -(c_n + c_s + c_w + c_e) + 1j * omega * mu * sigma_hat[k, j]
            
            # 添加矩阵元素
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j, k-1), c_n)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j, k+1), c_s)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j-1, k), c_w)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j+1, k), c_e)
            add_sparse_entry(rows, cols, data, idx, idx, c_center)
    
    return csr_matrix((data, (rows, cols)), shape=(n, n))

def assemble_tm_matrix(grid, sigma_hat, omega, mu=np.pi*4e-7, eps0=8.854e-12):
    """
    组装TM模式五点差分刚度矩阵
    
    对应Fortran MT2DTM子程序
    
    方程: ∇·(1/σ*∇Hₓ) + iωμHₓ = 0
    """
    n = grid.nnodes
    rows, cols, data = [], [], []
    
    # 计算sigma_y和sigma_z（方向平均）
    sigma_y = np.zeros((grid.nz + 1, grid.nx + 1), dtype=complex)
    sigma_z = np.zeros((grid.nz + 1, grid.nx + 1), dtype=complex)
    
    for j in range(1, grid.nx):
        for k in range(1, grid.nz):
            sigma_y[k, j] = (sigma_hat[k, j] * grid.dz[k] + 
                             sigma_hat[k-1, j] * grid.dz[k-1]) / (grid.dz[k] + grid.dz[k-1])
            sigma_z[k, j] = (sigma_hat[k, j] * grid.dy[j] + 
                             sigma_hat[k, j-1] * grid.dy[j-1]) / (grid.dy[j] + grid.dy[j-1])
    
    for k in range(1, grid.nz):
        for j in range(1, grid.nx):
            idx = grid.node_index(j, k)
            
            # TM模式差分系数
            coef_n = 2.0 / ((sigma_z[k-1, j] - 1j * omega * eps0) * 
                            grid.dz[k-1] * (grid.dz[k-1] + grid.dz[k]))
            coef_s = 2.0 / ((sigma_z[k, j] - 1j * omega * eps0) * 
                            grid.dz[k] * (grid.dz[k-1] + grid.dz[k]))
            coef_w = 2.0 / ((sigma_y[k, j-1] - 1j * omega * eps0) * 
                            grid.dy[j-1] * (grid.dy[j-1] + grid.dy[j]))
            coef_e = 2.0 / ((sigma_y[k, j] - 1j * omega * eps0) * 
                            grid.dy[j] * (grid.dy[j-1] + grid.dy[j]))
            
            # 中心系数
            c_center = -(coef_n + coef_s + coef_w + coef_e) + 1j * omega * mu
            
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j, k-1), coef_n)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j, k+1), coef_s)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j-1, k), coef_w)
            add_sparse_entry(rows, cols, data, idx, grid.node_index(j+1, k), coef_e)
            add_sparse_entry(rows, cols, data, idx, idx, c_center)
    
    return csr_matrix((data, (rows, cols)), shape=(n, n))
```

### 4.5 边界条件 (`core/boundary.py`)

```python
import numpy as np
from scipy.sparse import csr_matrix

def apply_te_bc(A, b, grid, Ex_value=1.0):
    """
    施加TE模式边界条件
    
    对应Fortran lines 61-123:
    - 上边界: Ex = 1
    - 左边界: Ex(K,1) - Ex(K,2) = 0  (∂Ex/∂x = 0)
    - 右边界: Ex(K,nx) - Ex(K,nx+1) = 0
    - 下边界: 混合边界
    """
    n = grid.nnodes
    
    # 上边界 (k=0)
    for j in range(1, grid.nx):
        idx = grid.node_index(j, 0)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        b[idx] = Ex_value
    
    # 左边界 (j=0)
    for k in range(1, grid.nz):
        idx = grid.node_index(0, k)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(1, k)] = -1.0
        b[idx] = 0
    
    # 右边界 (j=nx)
    for k in range(1, grid.nz):
        idx = grid.node_index(grid.nx, k)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(grid.nx - 1, k)] = -1.0
        b[idx] = 0
    
    # 下边界 (k=nz)
    for j in range(1, grid.nx):
        idx = grid.node_index(j, grid.nz)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(j, grid.nz - 1)] = -1.0
        b[idx] = 0

def apply_tm_bc(A, b, grid, Hx_value=0.0):
    """
    施加TM模式边界条件
    
    对应Fortran lines 208-270:
    - 上边界: Hx = 0
    - 左/右边界: ∂Hx/∂x = 0
    - 下边界: 混合边界
    """
    n = grid.nnodes
    
    # 上边界
    for j in range(1, grid.nx):
        idx = grid.node_index(j, 0)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        b[idx] = Hx_value
    
    # 左边界
    for k in range(1, grid.nz):
        idx = grid.node_index(0, k)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(1, k)] = -1.0
        b[idx] = 0
    
    # 右边界
    for k in range(1, grid.nz):
        idx = grid.node_index(grid.nx, k)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(grid.nx - 1, k)] = -1.0
        b[idx] = 0
    
    # 下边界
    for j in range(1, grid.nx):
        idx = grid.node_index(j, grid.nz)
        A[idx, :] = 0
        A[idx, idx] = 1.0
        A[idx, grid.node_index(j, grid.nz - 1)] = -1.0
        b[idx] = 0
```

### 4.6 pypardiso求解器 (`core/solver.py`)

```python
from pypardiso.scipy_solver import PardisoSPSolver
from scipy.sparse import csc_matrix

class ComplexPardisoSolver:
    """
    pypardiso复数矩阵求解器
    
    封装Intel MKL PARDISO求解器
    """
    
    def __init__(self, A):
        """
        初始化求解器
        
        参数:
            A: 稀疏矩阵 (scipy.sparse)
        """
        self.A = csc_matrix(A)
        self.solver = PardisoSPSolver(self.A, factorized=True)
    
    def solve(self, b):
        """
        求解 Ax = b
        
        参数:
            b: 右端向量 (numpy.ndarray)
        
        返回:
            x: 解向量 (numpy.ndarray)
        """
        return self.solver.solve(b)
    
    def free(self):
        """释放内存"""
        del self.solver
```

### 4.7 阻抗计算 (`core/impedance.py`)

```python
import numpy as np

def compute_te_impedance(Ex, grid, terrain, omega, mu=np.pi*4e-7):
    """
    计算TE模式阻抗 Zxy = Ex / Hy
    
    对应Fortran lines 162-168:
    Hy1_2 = (Ex(n+cols+1)-Ex(n)+Ex(n+cols+2)-Ex(n+1))/dz/(i*ω*μ)/2
    Ex1_2 = (Ex(n+cols+1)+Ex(n)+Ex(n+cols+2)+Ex(n+1))/4
    Zxy = Ex1_2 / Hy1_2
    """
    core = grid.core_cols
    Zxy = np.zeros(core, dtype=complex)
    
    for j in range(core):
        jn = j + grid.pad
        n = grid.node_index(jn, terrain[jn])
        
        # 中心差分求Hy
        Ex_avg = (Ex[n] + Ex[n + grid.nx + 1] + 
                  Ex[n + grid.nx + 2] + Ex[n + 1]) / 4
        Hy = (Ex[n + grid.nx + 1] - Ex[n] + 
               Ex[n + grid.nx + 2] - Ex[n + 1]) / grid.dz[terrain[jn] - 1] / (1j * omega * mu) / 2
        
        Zxy[j] = Ex_avg / Hy
    
    return np.conj(Zxy)

def compute_tm_impedance(Hx, grid, terrain, omega, sigma, mu=np.pi*4e-7, eps0=8.854e-12):
    """
    计算TM模式阻抗 Zyx = Ey / Hx
    
    对应Fortran lines 312-325:
    Ey = (Hx(n+cols+1)-Hx(n))/dz/(σ-i*ω*ε)
    Hx = (Hx(n+cols+1)+Hx(n))/2
    Zyx = Ey / Hx
    """
    core = grid.core_cols
    Zyx = np.zeros(core, dtype=complex)
    
    for j in range(core):
        jn = j + grid.pad
        n = grid.node_index(jn, terrain[jn])
        
        sig = sigma[jn, terrain[jn]] - 1j * omega * eps0
        
        Ey = (Hx[n + grid.nx + 1] - Hx[n]) / grid.dz[terrain[jn] - 1] / sig
        Hx_avg = (Hx[n + grid.nx + 1] + Hx[n]) / 2
        
        Zyx[j] = Ey / Hx_avg
    
    return np.conj(Zyx)

def apparent_resistivity(Z, omega, mu=np.pi*4e-7):
    """
    计算视电阻率和相位
    
    参数:
        Z: 复阻抗 (numpy.ndarray)
        omega: 角频率 (rad/s)
        mu: 磁导率 (H/m)
    
    返回:
        rho_a: 视电阻率 (Ω·m)
        phi: 相位 (度)
    """
    rho_a = np.abs(Z) ** 2 / (omega * mu)
    phi = np.angle(Z, deg=True)
    return rho_a, phi
```

## 5. 正演主程序 (`forward/modeling.py`)

```python
import numpy as np
from core.cole_cole import ColeColeSingle
from core.mesh import RectGrid256
from core.fd_operator import assemble_te_matrix, assemble_tm_matrix
from core.boundary import apply_te_bc, apply_tm_bc
from core.solver import ComplexPardisoSolver
from core.impedance import (compute_te_impedance, 
                            compute_tm_impedance,
                            apparent_resistivity)

def rfmt_forward(sigma, terrain, frequencies, dy, dz, 
                sigma0=0.001, m=0.1, tau=1e-3, c=0.5, 
                epsilon_r=5.0, pad=20):
    """
    RMT二维Cole-Cole正演主程序
    
    参数:
        sigma: 256×256 电导率数组 (S/m)
        terrain: 256 地形数组（每列地表行号，1-indexed）
        frequencies: nf 频率数组 (Hz)
        dy: 256 变步长x方向 (m)
        dz: 256 变步长z方向 (m)
        sigma0: Cole-Cole静态电导率 (S/m)
        m: Cole-Cole极化强度
        tau: Cole-Cole时间常数 (s)
        c: Cole-Cole指数
        epsilon_r: 相对介电常数
        pad: 边界padding层数
    
    返回:
        results: (nfreq * ncore_cols, 7) 数组
                 [fre, y, z, rhoxy, rhoyx, phasexy, phaseyx]
    """
    grid = RectGrid256(nx=256, nz=256, pad=pad)
    mu = np.pi * 4e-7
    eps0 = 8.854e-12
    
    # Cole-Cole模型
    cc = ColeColeSingle(sigma0=sigma0, m=m, tau=tau, c=c)
    
    # 核心区索引
    core_j = range(pad, 256 - pad)  # 216个
    
    results = []
    
    for f in frequencies:
        omega = 2 * np.pi * f
        
        # Cole-Cole + 位移电流
        sigma_cc = cc.sigma_total(omega, epsilon_r)
        sigma_full = sigma_cc * np.ones_like(sigma)
        
        # 变步长加权平均
        sigma_hat = grid.sigma_hat(sigma_full)
        
        # ========== TE模式 ==========
        A_te = assemble_te_matrix(grid, sigma_hat, omega, mu)
        b_te = np.zeros(grid.nnodes)
        apply_te_bc(A_te, b_te, grid, Ex_value=1.0)
        
        solver_te = ComplexPardisoSolver(A_te)
        Ex = solver_te.solve(b_te)
        solver_te.free()
        
        Zxy = compute_te_impedance(Ex, grid, terrain, omega, mu)
        rho_xy, phi_xy = apparent_resistivity(Zxy, omega, mu)
        
        # ========== TM模式 ==========
        A_tm = assemble_tm_matrix(grid, sigma_hat, omega, mu, eps0)
        b_tm = np.zeros(grid.nnodes)
        apply_tm_bc(A_tm, b_tm, grid, Hx_value=0.0)
        
        solver_tm = ComplexPardisoSolver(A_tm)
        Hx = solver_tm.solve(b_tm)
        solver_tm.free()
        
        Zyx = compute_tm_impedance(Hx, grid, terrain, omega, sigma, mu, eps0)
        rho_yx, phi_yx = apparent_resistivity(Zyx, omega, mu)
        
        # ========== 提取核心区结果 ==========
        for j in core_j:
            jn = j
            k = terrain[jn] - 1  # Fortran 1-indexed
            y = grid.y_nodes[jn]
            z = grid.z_nodes[k]
            
            j_idx = j - pad
            results.append([f, y, z, 
                          rho_xy[j_idx], rho_yx[j_idx],
                          phi_xy[j_idx], phi_yx[j_idx]])
    
    return np.array(results)

def save_results(results, filename='result_core.txt'):
    """
    保存核心区结果为文本文件
    
    参数:
        results: 结果数组
        filename: 输出文件名
    """
    header = ("fre(Hz)        y(m)           z(m)           "
              "rhoxy(Ω·m)   rhoyx(Ω·m)   phasexy(°)    phaseyx(°)")
    
    np.savetxt(filename, results,
                fmt='%.6e %.6e %.6e %.6e %.6e %.6e %.6e',
                header=header,
                comments='# ')
```
